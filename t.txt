AGWAccessLogs
| where ipv4_is_in_range(ClientIP, "") // Replace with Akamai CIDR
| project TimeGenerated, ClientIP, RequestUri, HttpStatusCode, Host, UserAgent
| order by TimeGenerated desc


NTANetAnalytics
| where FlowType == "ExternalPublic"
| where FlowStatus == "D" // "D" stands for Denied/Rejected
| project TimeGenerated, SrcIp, DestIp, DestPort, NSGRule, FlowStatus
| order by TimeGenerated desc

AzureDiagnostics
| where ResourceType == "NETWORKSECURITYGROUPS"
| where Action_s == "Deny"
| project TimeGenerated, SourceIP_s, DestIP_s, DestPort_d, Access_s, RuleName_s
| order by TimeGenerated desc
